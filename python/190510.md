


matplotlib으로 그래프 그리기

from matplotlib import pyplot as plt    # matplotlib 라이브러리 임폴트
import numpy as np

x=np.arange(1,10)  # 1부터 10까지의 범위
y=x*5   # 그래프

plt.plot(x,y)
plt.xlabel('x axis')  # x축 이름
plt.ylabel('y axis')  # y축 이름
plt.title('matplotlib sample')
plt.show() # 출력



위 의 경우 y=5x  그래프를 출력
